.Example: SFTP Listener (On New File Trigger)
[source,xml, linenums]
----
<flow name="onNewFile">
    <sftp:listener config-ref="file" directory="test-data/in" autoDelete="true">
      <scheduling-strategy>
        <fixed-frequency frequency="1000"/>
      </scheduling-strategy>
    </sftp:listener>

    <flow-ref name="processFile" />
</flow>
----
